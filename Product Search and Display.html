const productContainer = document.getElementById("product-list-container");

// Add click event listener to each product card element
productContainer.addEventListener("click", (event) => {
  const productCard = event.target.closest(".product-card"); // get the clicked product card element
  if (!productCard) return; // do nothing if the clicked element is not a product card

  const productId = productCard.getAttribute("data-id"); // get the id attribute of the product card element
  const selectedProduct = globalDataRecord.find((item) => item.id === parseInt(productId)); // find the product in the global data record using its id
  
  // Open a new page and display the selected product
  const newPage = window.open("", "_blank");
  newPage.document.write(`
    <h2>${selectedProduct.title}</h2>
    <img src="${selectedProduct.image}" alt="${selectedProduct.title}"/>
    <p>Price: Rs.${selectedProduct.price}</p>
    <p>Description: ${selectedProduct.description}</p>
  `);
});
